1. Without spcaes

shell.open("C:\\");

2. With spaces

shell.open("C:\\Program Files (x86)");

3. Double quotes inside single quotes

shell.open('"C:\\Program Files (x86)"');

4. Single quotes inside double quotes

shell.open("'C:\\Program Files (x86)'");

5. Escaping spaces

shell.open("C:\\Program\ Files\ (x86)");

6. Escaping spaces + double quotes inside single quotes

shell.open('"C:\\Program\ Files\ (x86)"');

6. Escaping spaces + single quotes inside double quotes

shell.open("'C:\\Program\ Files\ (x86)'");
